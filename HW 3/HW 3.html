# function to compute the probability of a single region having exactly k palindromes
pois_prob = function(lambda, k) {
    return( ((lambda^k) / factorial(k)) * (exp(1) ^ (-lambda)) )
}

# Create a function that splits the data into num_bins intervals, finds palindrome frequency within each interval, and calculates the p-value for a chi-squared test 
# for homogeneity

chisq_pvalue = function(num_bins){
    # count frequencies within each interval
    freqs = as.data.frame(table(cut(dna$location, breaks=seq(0, DNA_LENGTH, by=DNA_LENGTH/num_bins))))
    # calculate expected frequency of interval from total number of palindromes
    expected = NUM_PALINDROMES/nrow(freqs)
    freqs$expected = expected
    # calculate (O-E)^2/E
    freqs$stat = ((freqs$Freq - freqs$expected) ^ 2) / freqs$expected
    # sum together to find the chi-squared statistic
    chisq_stat = sum(freqs$stat)
    # calculate p-value from chi-squared distribution
    return (pchisq(chisq_stat, df=nrow(freqs)-1, lower.tail=F))
}

Biggest Cluster¶

Theoretical Approach:¶

Assume the null hypothesis, i.e. that palindromes occur according to a homogeneous Poisson process rather than clustering. When split into intervals of arbitrary size, we can count the number of palindromes in the most dense interval (the interval with the most occurences). Call this count of palindromes in the biggest cluster X. Below is a function, pvalue, that finds the probability that X is larger than a specified size i, i.e. P(X $\geq$ i), given a rate parameter and interval count.

In [20]:
pvalue = function(lambda, stat, num_intervals){
    cd = 0
    for (k in 0:stat-1){
        cd = cd + dpois(k, lambda)
    }
    return (1 - (cd^num_intervals))
}
